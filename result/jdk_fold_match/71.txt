Token processLookahead() throws ParseException {
  this.next();
  Token tok=Token.createLook(Token.LOOKAHEAD,this.parseRegex());
  if (this.read() != T_RPAREN)   throw ex("parser.factor.1",this.offset - 1);
  this.next();
  return tok;
}

}
