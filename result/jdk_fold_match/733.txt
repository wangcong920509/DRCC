private void painticonFocused(Graphics2D g){
  ellipse=decodeEllipse4();
  g.setPaint(color12);
  g.fill(ellipse);
  ellipse=decodeEllipse1();
  g.setPaint(decodeGradient3(ellipse));
  g.fill(ellipse);
  ellipse=decodeEllipse2();
  g.setPaint(decodeGradient4(ellipse));
  g.fill(ellipse);
}
