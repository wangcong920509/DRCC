private int retrieveIndex(long val){
  int low=0;
  int cursor=(int)val;
  if (varList == null || varList.length < 1)   return nbChildren;
  int max=varList.length - 1;
  int curr=low + (max - low) / 2;
  int elmt=0;
  while (low <= max) {
    elmt=varList[curr];
    if (cursor == elmt) {
      return curr;
    }
    if (elmt < cursor) {
      low=curr + 1;
    }
 else {
      max=curr - 1;
    }
    curr=low + (max - low) / 2;
  }
  return -1;
}
