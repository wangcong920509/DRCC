/** 
 * Constructor KeyInfo
 * @param doc
 */
public KeyInfo(Document doc){
  super(doc);
  XMLUtils.addReturnToElement(this._constructionElement);
}
/** 
 * Method add
 * @param keyvalue
 */
public void add(KeyValue keyvalue){
  this._constructionElement.appendChild(keyvalue.getElement());
  XMLUtils.addReturnToElement(this._constructionElement);
}
