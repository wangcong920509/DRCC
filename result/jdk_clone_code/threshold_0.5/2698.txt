private ByteBuffer putInt(long a,int x){
  if (unaligned) {
    int y=(x);
    unsafe.putInt(a,(nativeByteOrder ? y : Bits.swap(y)));
  }
 else {
    Bits.putInt(a,x,bigEndian);
  }
  return this;
}

  }
 else {
    Bits.putLong(a,x,bigEndian);
  }
  return this;
}
