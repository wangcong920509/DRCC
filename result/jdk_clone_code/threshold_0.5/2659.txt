private ByteBuffer putShort(long a,short x){
  if (unaligned) {
    short y=(x);
    unsafe.putShort(a,(nativeByteOrder ? y : Bits.swap(y)));
  }
 else {
    Bits.putShort(a,x,bigEndian);
  }
  return this;
}
